38   /**
39    * This is NOT an opmode.
40    *
41    * This class can be used to define all the specific hardware for a single robot.
42    * In this case that robot is a Pushbot.
43    * See PushbotTeleopTank_Iterative and others classes starting with "Pushbot" for usage examples.
44    *
45    * This hardware class assumes the following device names have been configured on the robot:
46    * Note:  All names are lower case and some have single spaces between words.
47    *
48    * Motor channel:  Left  drive motor:        "left_drive"
49    * Motor channel:  Right drive motor:        "right_drive"
50    * Motor channel:  Manipulator drive motor:  "left_arm"
51    * Servo channel:  Servo to open left claw:  "left_hand"
52    * Servo channel:  Servo to open right claw: "right_hand"
53    */
54   public class HardwarePushbot
55   {
56       /* Public OpMode members. */
57       public DcMotor fright = null;
58       public DcMotor fleft = null;
59       public DcMotor bright = null;
60       public DcMotor bleft = null;
61   
62       public static final double MID_SERVO       =  0.5 ;
63       public static final double ARM_UP_POWER    =  0.45 ;
64       public static final double ARM_DOWN_POWER  = -0.45 ;
65   
66       /* local OpMode members. */
67       HardwareMap hwMap           =  null;
68       private ElapsedTime period  = new ElapsedTime();
69   
70       /* Constructor */
71       public HardwarePushbot(){
72   
73       }
74   
75       /* Initialize standard Hardware interfaces */
76       public void init(HardwareMap ahwMap) {
77           // Save reference to Hardware map
78           hwMap = ahwMap;
79   
80           // Define and Initialize Motors
81           fright  = hwMap.get(DcMotor.class, "fright");
82           fleft = hwMap.get(DcMotor.class, "fleft");
83           bright = hwMap.get(DcMotor.class, "bright");
84           bleft = hwMap.get(DcMotor.class, "bleft");
85   
86           //leftArm    = hwMap.get(DcMotor.class, "left_arm");
87           fright.setDirection(DcMotor.Direction.FORWARD); // Set to REVERSE if using AndyMark motors
88           fleft.setDirection(DcMotor.Direction.FORWARD);// Set to FORWARD if using AndyMark motors
89           bright.setDirection(DcMotor.Direction.FORWARD);
90           bleft.setDirection(DcMotor.Direction.FORWARD);
91   
92           // Set all motors to zero power
93           fright.setPower(0);
94           fleft.setPower(0);
95           bright.setPower(0);
96           bleft.setPower(0);
97   
98           // Set all motors to run without encoders.
99           // May want to use RUN_USING_ENCODERS if encoders are installed.
100          fright.setMode(DcMotor.RunMode.RUN_USING_ENCODER);
101          fleft.setMode(DcMotor.RunMode.RUN_USING_ENCODER);
102          bright.setMode(DcMotor.RunMode.RUN_USING_ENCODER);
103          bleft.setMode(DcMotor.RunMode.RUN_USING_ENCODER);
104  //        leftDrive.setMode(DcMotor.RunMode.RUN_WITHOUT_ENCODER);
105  //        rightDrive.setMode(DcMotor.RunMode.RUN_WITHOUT_ENCODER);
106  //        leftArm.setMode(DcMotor.RunMode.RUN_WITHOUT_ENCODER);
107  
108          // Define and initialize ALL installed servos.
109  //        leftClaw  = hwMap.get(Servo.class, "left_hand");
110  //        rightClaw = hwMap.get(Servo.class, "right_hand");
111  //        leftClaw.setPosition(MID_SERVO);
112  //        rightClaw.setPosition(MID_SERVO);
113      }
114   }
