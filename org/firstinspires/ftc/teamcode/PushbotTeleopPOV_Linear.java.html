30   package org.firstinspires.ftc.teamcode;
31   
32   import com.qualcomm.robotcore.eventloop.opmode.Disabled;
33   import com.qualcomm.robotcore.eventloop.opmode.LinearOpMode;
34   import com.qualcomm.robotcore.eventloop.opmode.TeleOp;
35   import com.qualcomm.robotcore.util.Range;
36   
37   import org.firstinspires.ftc.robotcontroller.external.samples.HardwarePushbot;
38   
39   /**
40    * This OpMode uses the common Pushbot hardware class to define the devices on the robot.
41    * All device access is managed through the HardwarePushbot class.
42    * The code is structured as a LinearOpMode
43    *
44    * This particular OpMode executes a POV Game style Teleop for a PushBot
45    * In this mode the left stick moves the robot FWD and back, the Right stick turns left and right.
46    * It raises and lowers the claw using the Gampad Y and A buttons respectively.
47    * It also opens and closes the claws slowly using the left and right Bumper buttons.
48    *
49    * Use Android Studios to Copy this Class, and Paste it into your team's code folder with a new name.
50    * Remove or comment out the @Disabled line to add this opmode to the Driver Station OpMode list
51    */
52   
53   @TeleOp(name="Pushbot: Teleop POV", group="Pushbot")
54   @Disabled
55   public class PushbotTeleopPOV_Linear extends LinearOpMode {
56   
57       /* Declare OpMode members. */
58       org.firstinspires.ftc.robotcontroller.external.samples.HardwarePushbot robot           = new HardwarePushbot();   // Use a Pushbot's hardware
59       double          clawOffset      = 0;                       // Servo mid position
60       final double    CLAW_SPEED      = 0.02 ;                   // sets rate to move servo
61   
62       @Override
63       public void runOpMode() {
64           double left;
65           double right;
66           double drive;
67           double turn;
68           double max;
69   
70           /* Initialize the hardware variables. 
71            * The init() method of the hardware class does all the work here 
72            */
73           robot.init(hardwareMap);
74   
75           // Send telemetry message to signify robot waiting;
76           telemetry.addData("Say", "Hello Driver");    //
77           telemetry.update();
78   
79           // Wait for the game to start (driver presses PLAY)
80           waitForStart();
81   
82           // run until the end of the match (driver presses STOP)
83           while (opModeIsActive()) {
84   
85               // Run wheels in POV mode (note: The joystick goes negative when pushed forwards, so negate it)
86               // In this mode the Left stick moves the robot fwd and back, the Right stick turns left and right.
87               // This way it's also easy to just drive straight, or just turn.
88               drive = -gamepad1.left_stick_y;
89               turn  =  gamepad1.right_stick_x;
90   
91               // Combine drive and turn for blended motion.
92               left  = drive + turn;
93               right = drive - turn;
94   
95               // Normalize the values so neither exceed +/- 1.0
96               max = Math.max(Math.abs(left), Math.abs(right));
97               if (max > 1.0)
98               {
99                   left /= max;
100                  right /= max;
101              }
102  
103              // Output the safe vales to the motor drives.
104              robot.leftDrive.setPower(left);
105              robot.rightDrive.setPower(right);
106  
107              // Use gamepad left & right Bumpers to open and close the claw
108              if (gamepad1.right_bumper)
109                  clawOffset += CLAW_SPEED;
110              else if (gamepad1.left_bumper)
111                  clawOffset -= CLAW_SPEED;
112  
113              // Move both servos to new position.  Assume servos are mirror image of each other.
114              clawOffset = Range.clip(clawOffset, -0.5, 0.5);
115              robot.leftClaw.setPosition(robot.MID_SERVO + clawOffset);
116              robot.rightClaw.setPosition(robot.MID_SERVO - clawOffset);
117  
118              // Use gamepad buttons to move arm up (Y) and down (A)
119              if (gamepad1.y)
120                  robot.leftArm.setPower(robot.ARM_UP_POWER);
121              else if (gamepad1.a)
122                  robot.leftArm.setPower(robot.ARM_DOWN_POWER);
123              else
124                  robot.leftArm.setPower(0.0);
125  
126              // Send telemetry message to signify robot running;
127              telemetry.addData("claw",  "Offset = %.2f", clawOffset);
128              telemetry.addData("left",  "%.2f", left);
129              telemetry.addData("right", "%.2f", right);
130              telemetry.update();
131  
132              // Pace this loop so jaw action is reasonable speed.
133              sleep(50);
134          }
135      }
136  }
